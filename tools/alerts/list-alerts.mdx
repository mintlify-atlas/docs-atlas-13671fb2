## Overview

Retrieves a paginated list of all alerts associated with a specific incident. This tool is useful for understanding what triggered an incident and viewing all the alert details.

<Info>
This is a **read-only** tool. It retrieves data without modifying your PagerDuty account.
</Info>

## Parameters

<ParamField path="incident_id" type="string" required>
  The ID of the incident to retrieve alerts from.
  
  Example: `Q1XQPZ3PYQ0ABC`
</ParamField>

<ParamField path="query_model" type="AlertQuery" required>
  Query parameters for pagination and filtering.
  
  <Expandable title="AlertQuery Fields">
    <ParamField path="limit" type="integer" default="100">
      Maximum number of results to return. Must be between 1 and 1000.
      
      Default: `100`
    </ParamField>
    
    <ParamField path="offset" type="integer" default="0">
      Number of results to skip for pagination. Must be 0 or greater.
      
      Default: `0`
    </ParamField>
  </Expandable>
</ParamField>

## Response

Returns a `ListResponseModel[Alert]` containing an array of alerts.

<ResponseField name="response" type="Alert[]">
  Array of Alert objects associated with the incident.
  
  <Expandable title="Alert Object">
    <ResponseField name="id" type="string" required>
      The unique identifier for the alert.
    </ResponseField>
    
    <ResponseField name="type" type="string" required>
      The type of the object (always `"alert"`).
    </ResponseField>
    
    <ResponseField name="summary" type="string" required>
      A short summary of the alert.
    </ResponseField>
    
    <ResponseField name="self" type="string" required>
      The API URL at which this alert is accessible.
    </ResponseField>
    
    <ResponseField name="html_url" type="string" required>
      The URL at which this alert is accessible in the PagerDuty UI.
    </ResponseField>
    
    <ResponseField name="created_at" type="datetime" required>
      The date/time when this alert was created.
    </ResponseField>
    
    <ResponseField name="status" type="string" required>
      The current status of the alert (e.g., `triggered`, `resolved`).
    </ResponseField>
    
    <ResponseField name="alert_key" type="string" required>
      The unique key identifying this alert.
    </ResponseField>
    
    <ResponseField name="service" type="ServiceReference">
      Reference to the service this alert is associated with.
      
      <Expandable title="ServiceReference">
        <ResponseField name="id" type="string">
          The ID of the service.
        </ResponseField>
        
        <ResponseField name="type" type="string">
          The type (always `"service_reference"`).
        </ResponseField>
        
        <ResponseField name="summary" type="string">
          A short summary of the service.
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="incident" type="IncidentReference">
      Reference to the incident this alert is associated with.
      
      <Expandable title="IncidentReference">
        <ResponseField name="id" type="string">
          The ID of the incident.
        </ResponseField>
        
        <ResponseField name="type" type="string">
          The type of the reference.
        </ResponseField>
        
        <ResponseField name="summary" type="string">
          A short summary of the incident.
        </ResponseField>
        
        <ResponseField name="self" type="string">
          The API URL for the incident.
        </ResponseField>
        
        <ResponseField name="html_url" type="string">
          The UI URL for the incident.
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="body" type="AlertBody">
      The detailed body of the alert.
      
      <Expandable title="AlertBody">
        <ResponseField name="type" type="string">
          The type of alert body.
        </ResponseField>
        
        <ResponseField name="contexts" type="array">
          Additional context information for the alert.
        </ResponseField>
        
        <ResponseField name="details" type="object">
          Detailed information about the alert.
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="severity" type="string">
      The severity level of the alert (e.g., `critical`, `error`, `warning`, `info`).
    </ResponseField>
    
    <ResponseField name="suppressed" type="boolean">
      Whether the alert is suppressed.
    </ResponseField>
    
    <ResponseField name="integration" type="IntegrationReference">
      Reference to the integration that created this alert.
      
      <Expandable title="IntegrationReference">
        <ResponseField name="id" type="string">
          The ID of the integration.
        </ResponseField>
        
        <ResponseField name="type" type="string">
          The type of the integration.
        </ResponseField>
        
        <ResponseField name="summary" type="string">
          A short summary of the integration.
        </ResponseField>
        
        <ResponseField name="self" type="string">
          The API URL for the integration.
        </ResponseField>
        
        <ResponseField name="html_url" type="string">
          The UI URL for the integration.
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="response_summary" type="string">
  A human-readable summary of the response including the count and type of records returned.
</ResponseField>
